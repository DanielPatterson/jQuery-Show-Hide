<p class="example">If you click on the "Hide" button, I will disappear.</p>
<a href="#" class="btn btn-md btn-default" id="hide">Hide</a>
<a href="#" class="btn btn-md btn-default" id="show">Show</a>

$(document).ready(function(){
    $("a#hide").click(function(){
        $("p.example").hide();
    });
    $("a#show").click(function(){
        $("p.example").show();
    });
});
